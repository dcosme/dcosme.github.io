Cleaning

  • exclude non-sub responses
  • convert missing values to NA
  • duplicates

First, exclude responses that are not subject responses. In this dataset, some subjects have their ID in the ExternalDataReference column only, so we’ll need to add that to the SID column before filtering. There are also some test responses that match our SID pattern, so we’ll want to exclude those using the exclude_SID pattern.

missingSIDs = surveys_long %>%
  mutate(SID = ifelse(is.na(SID), ExternalDataReference,
               ifelse(grepl("SV", SID), ExternalDataReference, SID))) %>%
  select(-ExternalDataReference) %>%
  filter(SID == "" | grepl(sid_pattern, SID)) %>%
  filter(!grepl(exclude_sid, SID)) %>%
  arrange(value) %>%
  filter(item == "Name") %>%
  mutate(SID = ifelse(SID == "" & survey_name == "CheckIn_Fall", lead(SID), SID)) %>%
  filter(!SID == "") %>%
  select(-item, -value)

t1 = surveys_long %>%
  filter(survey_name == "Freshman Project T1 Survey")

surveys_long_sub = surveys_long %>%
  filter(!survey_name == "Freshman Project T1 Survey") %>%
  select(-SID) %>%
  left_join(., missingSIDs, by = c("qid", "survey_name")) %>%
  bind_rows(t1) %>%
  filter(grepl(sid_pattern, SID)) %>%
  filter(!grepl(exclude_sid, SID)) %>%
  filter(!item == "Name" &!is.na(SID)) %>%
  select(-ExternalDataReference)

subset goals questions

gps = surveys_long_clean %>% 
  filter(grepl("GPS", item)) %>%
  filter(!item == "GPS_1") %>%
  mutate(term = ifelse(survey_name %in% c("CheckIn_Fall", "Freshman Project T2 Survey"), "Fall",
                ifelse(survey_name %in% c("CheckIn_Winter", "Freshman Project T3 Survey"), "Winter",
                ifelse(survey_name %in% c("CheckIn_Spring", "Freshman Project T4 Survey"), "Spring", NA))),
         survey = ifelse(grepl("CheckIn", survey_name), "check in", "follow up"),
         personalproject = ifelse(grepl("GPS_2_[1-3]{1}", item), "first", "second"),
         item = ifelse(grepl("GPS_[2-3]{1}_1", item), "importance",
                ifelse(grepl("GPS_[2-3]{1}_2", item), "commitment",
                ifelse(grepl("GPS_[2-3]{1}_3", item), "complete", item))),
         sort = ifelse(item == "importance", 1,
                ifelse(item == "commitement", 2, 3)))
